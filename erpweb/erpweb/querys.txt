ALTER PROCEDURE [dbo].[web_carga_nv_cab_web]
	@v_id_nv int,
	@v_nv int,
	@v_fecha datetime,
	@v_obs nvarchar(500),
	@v_Id_cliente int, 
	@v_id_contacto int,
	@v_Suma_total float, 
	@v_Neto_venta float, 
	@v_Tax_venta float, 
	@v_total float,
	@v_sigla_moneda nvarchar(3),
	@v_Direcciondespacho nvarchar(500), 
	@v_ciudaddespacho nvarchar(100), 
	@v_comunadespacho nvarchar(100), 
	@v_Folio_transac_web nvarchar(100),
	@v_id_vendedor int,
	@v_id_tipo_fact int
AS
BEGIN TRY
   BEGIN TRAN
   declare @cmd nvarchar(500)
   declare @id_cliente int
   declare @id_contacto int
   declare @ctrl int
   declare @id_cot int
   declare @id_nta_vta int
   declare @_id_nta_vta int
   declare @nta_vta_num int
   declare @rut int
   declare @dv nvarchar(1)
   declare @v_id_moneda int
   declare @_v_tipo_fact int
   declare @v_id_despachar int
	-- SET NOCOUNT ON added to prevent extra result sets from
	-- interfering with SELECT statements.
	SET NOCOUNT ON;

	select @v_id_moneda = ID_Moneda from tbl_monedas where sigla = replace(@v_sigla_moneda,' ','')

	select @_id_nta_vta = ID_Nta_Vta from tbl_Nota_Venta where id_nta_vta_web = @v_id_nv

	if (@v_id_tipo_fact = 1)
	begin
		set @_v_tipo_fact = 11	
	end
	else
	begin
	  set @_v_tipo_fact = 17	
	end

	-- sucursal de despacho
	SELECT TOP 1  @v_id_despachar=ID_Sucursal FROM  dbo.tbl_Sucursales_Clientes WHERE  (Id_Cliente = @v_Id_cliente) AND (Activo = 1) ORDER BY Nombre_Sucursal

	if isnull(@_id_nta_vta,0) = 0
	begin

		   select @nta_vta_num = Correlativo_NV + 1 from tbl_Correlativos

		   insert into tbl_Nota_Venta(Id_Nta_Vta_Web,
									 Nta_Vta_Num_Web, 
									 Fecha_Not_Vta,
									 Observaciones, 
									 Id_cliente, 
									 id_contacto,
									 id_vendedor,
									 Suma_total, 
									 Neto_venta, 
									 Tax_venta, 
									 total, 
									 Id_Moneda,
									 Direcciondespacho, 
									 ciudaddespacho, 
									 comunadespacho, 
									 Folio_transac_web,
									 nta_vta_num,
									 Id_Estado,
									 Id_Tipo_NV,
									 Nombre_Not_Vta,
									 Salida_con_Factura,
									 Salida_con_Guia,
									 Id_Bodega,
									 Forma_Pago,
									 Diascred,
									 Id_Filial,
									 Id_Tipo_Fact,
									 Id_despachar,
									 Id_Salida_Factura,
									 Id_Autor
									 )
		  Values (@v_id_nv,
		          @v_nv,
				  @v_fecha,
				  @v_obs,
				  @v_Id_cliente,
				  @v_id_contacto,
				  @v_id_vendedor,
				  @v_Suma_total, 
				  @v_Neto_venta, 
				  @v_Tax_venta, 
				  @v_total,
				  @v_id_moneda,
				  @v_Direcciondespacho, 
				  @v_ciudaddespacho, 
				  @v_comunadespacho, 
				  @v_Folio_transac_web,
				  @nta_vta_num,
				  1, -- Preparándose
			      6,
			      'Nv Web',
			      1,
			      0,
			      1, -- id de la Bodega Renca
			      'EF', -- forma pago
			      0, ---dias credito
			      1,
				  @v_id_tipo_fact,
				  @v_id_despachar,
				  9, --Correo Electrónico
				  @v_id_vendedor
				  )

		-- cabecera cotización
		-- detalle cotización
		update tbl_Correlativos set Correlativo_NV = Correlativo_NV + 1
		select @_id_nta_vta = id_nta_vta from tbl_Nota_Venta where Id_Nta_Vta_Web = @v_id_nv
	end
	select @_id_nta_vta
	COMMIT TRAN
END TRY

BEGIN CATCH
    ROLLBACK TRAN
    select cast(ERROR_LINE() as nvarchar)+ ' '+ cast(ERROR_NUMBER() as nvarchar) + ' ' + ERROR_MESSAGE()
END CATCH;
